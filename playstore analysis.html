def size_to_float(size):
     """
    Remove M and k from value, converts KB to MB
    Parameters:
     size: the size value to clean
      """
    if size.endswith('M'):
        return float(size.replace("M","").replace(",",""))
    elif size.endswith('k'):
        return float(size.replace("k","").replace(",",""))/1000
    else:
        return size

def plot_appcount(data_frame, year):
    """
    Plot number of apps per gategory for specific year
    Parameters:
      data_frame - Data frame.
      year - The year label
    """
    fig, ax = plt.subplots(1, 1, figsize=(10, 6))
    #data_frame.Category.value_counts().head(10).sort_values(ascending=True).plot(kind='bar')
    plt.title('# Installs per Category in ' + year,size = 14)
    data_frame.groupby('Category')['Installs'].sum().sort_values(ascending=True).plot(kind='bar')#.groups.keys()
