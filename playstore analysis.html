def size_to_float(size):
    """
    Remove M and k from input and converts KB to MB
    
    Parameter:
    size: string representation of size
    
    Returns:
    size: float
    """
    
    if size.endswith('M'):
        return float(size.replace("M","").replace(",",""))
    elif size.endswith('k'):
        return float(size.replace("k","").replace(",",""))/1000
    else:
        return size
